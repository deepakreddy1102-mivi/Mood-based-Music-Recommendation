# cleaning
df_clean = df.copy()

# filling numeric nulls
for c in num_cols:
    df_clean[c] = df_clean[c].fillna(df_clean[c].median())

# filling categorical nulls
if "Time of Day" in df_clean.columns:
    df_clean["Time of Day"] = df_clean["Time of Day"].fillna(df_clean["Time of Day"].mode()[0])
df_clean["Mood"] = df_clean["Mood"].fillna(df_clean["Mood"].mode()[0])

def iqr_cap(s, k=1.5):
    q1, q3 = s.quantile(0.25), s.quantile(0.75)
    iqr = q3 - q1
    low, high = q1 - k*iqr, q3 + k*iqr
    return s.clip(lower=low, upper=high)

for c in num_cols:
    df_clean[c] = iqr_cap(df_clean[c], k=1.5)

# Pre processing

from sklearn.compose import ColumnTransformer
from sklearn.preprocessing import OneHotEncoder, StandardScaler
from sklearn.pipeline import Pipeline
from sklearn.linear_model import LogisticRegression
from sklearn.tree import DecisionTreeClassifier
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier

num_features = [c for c in ["Heart Rate","Skin Temperature","Blink Rate","Score"] if c in X_train.columns]
cat_features = [c for c in ["Time of Day"] if c in X_train.columns]

preprocess = ColumnTransformer(
    transformers=[
        ("num", StandardScaler(), num_features),
        ("cat", OneHotEncoder(handle_unknown="ignore"), cat_features),
    ]
)

models = {
    "LogReg": LogisticRegression(max_iter=200),
    "DecisionTree": DecisionTreeClassifier(random_state=42),
    "RandomForest": RandomForestClassifier(n_estimators=300, random_state=42),
    "GradBoost": GradientBoostingClassifier(random_state=42),
}
print("numeric:", num_features, "| categorical:", cat_features)

from sklearn.model_selection import StratifiedKFold, cross_val_score
import pandas as pd

cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
scores = {}

for name, clf in models.items():
    pipe = Pipeline([("prep", preprocess), ("clf", clf)])
    acc = cross_val_score(pipe, X_train, y_train, cv=cv, scoring="accuracy")
    scores[name] = (acc.mean(), acc.std())

score_df = pd.DataFrame(scores, index=["mean_acc","std_acc"]).T.sort_values("mean_acc", ascending=False)

# Mood & Music mapping

MOOD_TO_GENRE = {
    "Happy":    "Pop",
    "Sad":      "Acoustic",
    "Stressed": "Lo-fi",
    "Relaxed":  "Ambient",
    "Energetic":"EDM"
}

FINAL_MODEL = search.best_estimator_ if "search" in globals() else pipe

def recommend_music(features: dict, model=FINAL_MODEL):
    """
    features example:
    {
      "Heart Rate": 78.0,
      "Skin Temperature": 36.4,
      "Blink Rate": 18.0,
      "Time of Day": "Afternoon",
      "Score": 12.3
    }
    """
    cols = [c for c in ["Heart Rate","Skin Temperature","Blink Rate","Time of Day","Score"] if c in X.columns]
    X_in = pd.DataFrame([features], columns=cols)
    pred_mood = model.predict(X_in)[0]
    return {"predicted_mood": pred_mood, "recommended_genre": MOOD_TO_GENRE.get(pred_mood, "Pop")}


demo = {"Heart Rate": 75.0, "Skin Temperature": 36.5, "Blink Rate": 16.0, "Time of Day": "Afternoon", "Score": 10.0}
